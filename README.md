## Install

```bash
npm install --save react-step-by-step
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-step-by-step'
import 'react-step-by-step/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
